11/// Builder for creating a hotkey listener.
12#[derive(Default)]
13pub struct HotkeyListenerBuilder {
14    hotkeys: Vec<Hotkey>,
15}
16
17impl HotkeyListenerBuilder {
18    /// Create a new builder.
19    pub fn new() -> Self {
20        Self::default()
21    }
22
23    /// Add a hotkey to listen for.
24    pub fn add_hotkey(mut self, hotkey: Hotkey) -> Self {
25        self.hotkeys.push(hotkey);
26        self
27    }
28
29    /// Build the listener.
30    #[cfg(target_os = "linux")]
31    pub fn build(self) -> Result<HotkeyListener> {
32        let keyboards = crate::linux::find_keyboards()?;
33        Ok(HotkeyListener {
34            inner: crate::linux::HotkeyListener::new(keyboards, self.hotkeys),
35        })
36    }
37
38    /// Build the listener.
39    #[cfg(target_os = "macos")]
40    pub fn build(self) -> Result<HotkeyListener> {
41        Ok(HotkeyListener {
42            inner: crate::macos::HotkeyListener::new(self.hotkeys),
43        })
44    }
45
46    /// Build the listener (unsupported platform stub).
47    #[cfg(not(any(target_os = "linux", target_os = "macos")))]
48    pub fn build(self) -> Result<HotkeyListener> {
49        anyhow::bail!("Hotkey listening is not supported on this platform")
50    }
51}

53/// A hotkey listener that runs in a background thread.
54pub struct HotkeyListener {
55    #[cfg(target_os = "linux")]
56    inner: crate::linux::HotkeyListener,
57    #[cfg(target_os = "macos")]
58    inner: crate::macos::HotkeyListener,
59    #[cfg(not(any(target_os = "linux", target_os = "macos")))]
60    inner: (),
61}
62
63impl HotkeyListener {
64    /// Start listening for hotkeys in a background thread.
65    ///
66    /// Returns a [`HotkeyListenerHandle`] that receives hotkey events.
67    /// The background thread automatically stops when the handle is dropped.
68    #[cfg(any(target_os = "linux", target_os = "macos"))]
69    pub fn start(self) -> Result<HotkeyListenerHandle> {
70        let running = Arc::new(AtomicBool::new(true));
71        let rx = self.inner.start(Arc::clone(&running))?;
72        Ok(HotkeyListenerHandle { running, rx })
73    }
74
75    /// Start listening (unsupported platform stub).
76    #[cfg(not(any(target_os = "linux", target_os = "macos")))]
77    pub fn start(self) -> Result<HotkeyListenerHandle> {
78        anyhow::bail!("Hotkey listening is not supported on this platform")
79    }
80}

82/// Handle for receiving hotkey events.
83///
84/// The background listener thread automatically stops when this handle is dropped,
85/// providing automatic cleanup without requiring manual shutdown signals.
86///
87/// # Platform Notes
88///
89/// On Linux, the listener thread exits immediately when the handle is dropped.
90/// On macOS, `rdev::listen()` receives all keyboard events system-wide and filters
91/// for registered hotkeys. The thread cannot be stopped early but terminates with
92/// the process.
93///
94/// # Example
95///
96/// ```no_run
97/// use hotkey_listener::{parse_hotkey, HotkeyListenerBuilder};
98/// use std::time::Duration;
99///
100/// let hotkey = parse_hotkey("F8").unwrap();
101/// let handle = HotkeyListenerBuilder::new()
102///     .add_hotkey(hotkey)
103///     .build()
104///     .unwrap()
105///     .start()
106///     .unwrap();
107///
108/// // Receive events
109/// while let Ok(event) = handle.recv_timeout(Duration::from_millis(100)) {
110///     println!("Event: {:?}", event);
111/// }
112///
113/// // Thread stops automatically when handle goes out of scope
114/// ```
115pub struct HotkeyListenerHandle {
116    running: Arc<AtomicBool>,
117    rx: Receiver<HotkeyEvent>,
118}
119
120impl HotkeyListenerHandle {
121    /// Block until the next hotkey event.
122    pub fn recv(&self) -> Result<HotkeyEvent, RecvError> {
123        self.rx.recv()
124    }
125
126    /// Wait for the next hotkey event with a timeout.
127    pub fn recv_timeout(&self, timeout: Duration) -> Result<HotkeyEvent, RecvTimeoutError> {
128        self.rx.recv_timeout(timeout)
129    }
130
131    /// Try to receive a hotkey event without blocking.
132    pub fn try_recv(&self) -> Result<HotkeyEvent, std::sync::mpsc::TryRecvError> {
133        self.rx.try_recv()
134    }
135
136    /// Check if the listener is still running.
137    pub fn is_running(&self) -> bool {
138        self.running.load(Ordering::SeqCst)
139    }
140
141    /// Manually stop the listener.
142    ///
143    /// This is called automatically when the handle is dropped.
144    pub fn stop(&self) {
145        self.running.store(false, Ordering::SeqCst);
146    }
147}
148
149impl Drop for HotkeyListenerHandle {
150    fn drop(&mut self) {
151        self.running.store(false, Ordering::SeqCst);
152    }
153}
